# Conda

- Installation

Download from website and bash

- Create environment with specific python version

conda create -n [name for environment] python=[python version]

- Rename environment

conda rename -n [original name] [new name]

- Delete environment

conda remove -n [environment name] --all

- Activate an environment

conda activate [environment name]

- Deactivate an environment

conda deactivate

- Update all packages

conda update --all

- List all installed packages

conda list

- List all environments/locations

conda env list

- Install package

conda install [package name]

- Uninstall package

conda uninstall [package name]

- Remove all unused files

conda clean --all

# Git

- setting a name for identification

git config --global user.name "[firstname lastname]"

- setting an email address

git config --global user.email "[email]"

- initialize exisiting directory as Git repository

git init

- retriving an entire repository

git clone [url]

- Seeing modified directories

git status

- adding files

git add [file]

- commiting a modification

git commit -m "[message]"

- Transmit local branch commits to repository

git push [alias] [branch]
